import sys

def gcd(n, m):
    while (m != 0):
        n, m = m, n % m
    return n

for _ in range(int(input())):
    numbers = list(map(int, input().split()))
    numbers = numbers[1:]
    sum_gcd = []
    for i in range(len(numbers)):
        for j in range(i + 1, len(numbers)):
            GCD = gcd(numbers[i], numbers[j])
            sum_gcd.append(GCD)
    answer = sum(sum_gcd)
    print(answer)
